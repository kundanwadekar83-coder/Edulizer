तुमची प्रोफाइल

तुमची वैयक्तिक आणि शैक्षणिक माहिती येथे अपडेट ठेवा.

// प्रोफाइल अपडेट लॉजिक document.getElementById('profileForm').addEventListener('submit', function(e) { e.preventDefault(); user.name = document.getElementById('profName').value; user.category = document.getElementById('profCat').value; user.rank = document.getElementById('profRank').value; localStorage.setItem('currentUser', JSON.stringify(user)); // मूळ डेटाबेस (localstorage index) देखील अपडेट करणे localStorage.setItem('user_' + user.email, JSON.stringify(user)); alert('माहिती अपडेट झाली!'); location.reload(); }); function logout() { localStorage.removeItem('currentUser'); window.location.href = 'login.html'; }
